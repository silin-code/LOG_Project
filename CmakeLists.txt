cmake_minimum_required(VERSION 3.20)
project(log_test)

# 强制使用C++11（你的代码必须依赖这个）
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 关键：告诉编译器，头文件都在logs文件夹里！
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/logs)

# 把所有文件都加入构建，VS能自动识别
add_executable(log_test 
    test.cpp
    logs/format.hpp
    logs/level.hpp
    logs/message.hpp
    logs/util.hpp
    logs/sink.hpp
    logs/looper.hpp
    logs/buffer.hpp
    logs/log.hpp)

# VS2022兼容设置（解决乱码和警告）
if(MSVC)
    add_compile_options(/utf-8)
    target_compile_definitions(log_test PRIVATE _CRT_SECURE_NO_WARNINGS)
endif()
